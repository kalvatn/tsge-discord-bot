channel
  id <pk> <int>
  name <string>

user
  id <pk> <int>
  username <key?> <string>
  nick <string>

user_channel_message
  channel_id <fk channel.id>
  user_id  <fk user.id>
  contents <string|text?>

user_channel_stats_type
  id <pk> <int>
  name <string>
  value_type <string|enum?> (int, string, ...)

user_channel_stats
  channel_id <fk channel.id>
  user_id <fk user.id>
  user_channel_stats_type_id <fk user_channel_stats_type.id>
  value <string?>

SELECT
  u.id AS user_id
  , ucst.id AS stat_type_id, ucst.name AS stat_type_name, ucst.type AS stat_type_value_type
  , ucs.value
FROM user u
  LEFT JOIN user_channel_stats ucs ON ucs.user_id = u.id
  LEFT JOIN user_channel_stats_type ucst ON ucst.id = ucs.user_channel_stats_type_id
WHERE 1
  AND u.id = <user_id>
  AND ucs.channel_id = <channel_id>
  AND ucst.id IN (1, 2, 3, 4)
;
